**文件：**

```typescript
import { describe, it, expect } from 'vitest'
import { createTransport } from '../../src/rpc/transport'
import { Readable, Writable } from 'stream'

describe('JSON-RPC Transport', () => {
  it('parses a JSON-RPC request from input stream', async () => {
    const input = Readable.from(['{"jsonrpc":"2.0","id":1,"method":"ping","params":{}}\n'])
    const output = new Writable({ write() {} })
    const transport = createTransport(input, output)

    const request = await transport.readRequest()

    expect(request).toEqual({ jsonrpc: '2.0', id: 1, method: 'ping', params: {} })
  })

  it('writes a JSON-RPC response to output stream', async () => {
    const input = Readable.from([])
    const chunks: string[] = []
    const output = new Writable({
      write(chunk, _, cb) { chunks.push(chunk.toString()); cb() }
    })
    const transport = createTransport(input, output)

    await transport.writeResponse({ jsonrpc: '2.0', id: 1, result: { ok: true } })

    expect(JSON.parse(chunks.join(''))).toEqual({ jsonrpc: '2.0', id: 1, result: { ok: true } })
  })

  it('writes a JSON-RPC notification (no id)', async () => {
    const input = Readable.from([])
    const chunks: string[] = []
    const output = new Writable({
      write(chunk, _, cb) { chunks.push(chunk.toString()); cb() }
    })
    const transport = createTransport(input, output)

    await transport.writeNotification('chat/chunk', { content: 'hi' })

    expect(JSON.parse(chunks.join(''))).toEqual({ jsonrpc: '2.0', method: 'chat/chunk', params: { content: 'hi' } })
  })
})
```


- [ ] **Green: 最小实现**

实现 `createTransport(input, output)` 返回对象：
- `readRequest()`: 按行读取 stdin，JSON.parse 每行
- `writeResponse(resp)`: JSON.stringify + '\n' 写入 stdout
- `writeNotification(method, params)`: 同上，无 id 字段

```typescript
import { describe, it, expect } from 'vitest'
import { spawn } from 'child_process'

function sendRpc(proc: any, method: string, params: any, id: number): Promise<any> {
  return new Promise((resolve) => {
    const handler = (data: Buffer) => {
      const lines = data.toString().split('\n').filter(Boolean)
      for (const line of lines) {
        const msg = JSON.parse(line)
        if (msg.id === id) {
          proc.stdout.off('data', handler)
          resolve(msg)
        }
      }
    }
    proc.stdout.on('data', handler)
    proc.stdin.write(JSON.stringify({ jsonrpc: '2.0', id, method, params }) + '\n')
  })
}

describe('E2E', () => {

    const proc = spawn('npx', ['tsx', 'src/index.ts'], { stdio: ['pipe', 'pipe', 'pipe'] })

    }
```
